Cloudflare Workers Python 3.12 CI/CD Passing KV Storage Queues
cloudflare-python-workers — demo.py
start demo servers
PS> python demo.py
# api-worker -> http://localhost:8001
# task-producer -> http://localhost:8002
# task-consumer -> background thread (watching queue)
api-worker — CRUD over KV
PS> Invoke-RestMethod http://localhost:8001/items
keys : {}  count : 0
PS> Invoke-RestMethod -Method POST http://localhost:8001/items/hello -Body '"world"'
ok : True  key : hello
PS> Invoke-RestMethod http://localhost:8001/items/hello
key : hello  value : world
PS> Invoke-RestMethod http://localhost:8001/items
keys : {hello}  count : 1
task-producer + task-consumer — queue flow
PS> Invoke-RestMethod -Method POST http://localhost:8002/enqueue \
    -Body '{"type":"store","key":"job-result","value":"done"}'
ok : True  job_id : f692b762
# 500ms later — consumer processes the job...
PS> Invoke-RestMethod http://localhost:8001/items/job-result
key : job-result  value : done
PS> Invoke-RestMethod -Method POST http://localhost:8002/enqueue \
    -Body '{"type":"delete","key":"job-result"}'
ok : True  job_id : aafaaf66
PS> Invoke-RestMethod http://localhost:8001/items
keys : {}  count : 0
PS> Invoke-RestMethod -Method DELETE http://localhost:8001/items/hello
ok : True  key : hello
